<!-- Directiva define:vars inyecta la data del JSON serializada de forma segura en el cliente, logrando Zero-Server-Load -->
<script define:vars={{ productosCliente: productos }}>
  // Vanilla JS puro: Cero dependencias para mantener TBT en 0ms.
  document.addEventListener('DOMContentLoaded', () => {
    const select1 = document.getElementById('producto-1');
    const select2 = document.getElementById('producto-2');
    const contenedorTabla = document.getElementById('tabla-comparativa');

    // API nativa de internacionalización para formato monetario
    function formatearPrecio(precio) {
      return new Intl.NumberFormat('es-MX', { style: 'currency', currency: 'MXN' }).format(precio);
    }

    function renderizarComparacion() {
      const id1 = select1.value;
      const id2 = select2.value;

      // Si falta uno de los dos por seleccionar, detenemos la ejecución
      if (!id1 || !id2) return;

      // Búsqueda en memoria local (ultrarrápida) O(N)
      const p1 = productosCliente.find(p => p.id === id1);
      const p2 = productosCliente.find(p => p.id === id2);

      if (p1 && p2) {
        // Unificamos las llaves de especificaciones utilizando un Set para evitar duplicidad de specs
        const specsKeys = [...new Set([
          ...Object.keys(p1.especificaciones), 
          ...Object.keys(p2.especificaciones)
        ])];

        // Construcción manual de un string HTML semántico
        let html = `
          <div class="tabla-scroll">
            <table class="kp-tabla-comparativa">
              <thead>
                <tr>
                  <th scope="col" style="width: 30%">Métrica</th>
                  <th scope="col" style="width: 35%">${p1.nombre}</th>
                  <th scope="col" style="width: 35%">${p2.nombre}</th>
                </tr>
              </thead>
              <tbody>
                <tr class="fila-destacada">
                  <td class="spec-label">Precio Transaccional</td>
                  <td class="precio-celda">${formatearPrecio(p1.precio_mxn)}</td>
                  <td class="precio-celda">${formatearPrecio(p2.precio_mxn)}</td>
                </tr>
        `;

        // Iteramos sobre las llaves combinadas de hardware
        specsKeys.forEach(key => {
          const val1 = p1.especificaciones[key];
          const val2 = p2.especificaciones[key];
          
          const cleanVal1 = (!val1 || val1 === 'No especificado') ? '<span class="empty-spec">—</span>' : val1;
          const cleanVal2 = (!val2 || val2 === 'No especificado') ? '<span class="empty-spec">—</span>' : val2;

          html += `
            <tr>
              <td class="spec-label">${key.replace(/_/g, ' ')}</td>
              <td>${cleanVal1}</td>
              <td>${cleanVal2}</td>
            </tr>
          `;
        });

        // Fila de cierre con CTAs dinámicos apuntando al Cloaker perimetral de Cloudflare (/recomienda/)
        // Inyectando obligatoriamente rel="nofollow noopener noreferrer"
        html += `
                <tr class="fila-cta">
                  <td></td>
                  <td>
                    <a href="/recomienda/${p1.id}" rel="nofollow noopener noreferrer" target="_blank" class="cta-comparador">
                      Ver Disponibilidad
                    </a>
                  </td>
                  <td>
                    <a href="/recomienda/${p2.id}" rel="nofollow noopener noreferrer" target="_blank" class="cta-comparador">
                      Ver Disponibilidad
                    </a>
                  </td>
                </tr>
              </tbody>
            </table>
          </div>
        `;

        // Inyección cruda y segura al DOM
        contenedorTabla.innerHTML = html;
      }
    }

    // Escuchadores de eventos pasivos
    select1.addEventListener('change', renderizarComparacion);
    select2.addEventListener('change', renderizarComparacion);
  });
</script>

import fs from 'node:fs';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

// 1. Configuración nativa de rutas para Node.js (ES Modules)
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

// Definimos de forma estricta las rutas absolutas para lectura y escritura
const dataPath = path.resolve(__dirname, '../data/productos.json');
const outputDir = path.resolve(__dirname, '../../public');
const outputPath = path.resolve(outputDir, '_redirects');

// 2. Proceso Seguro en Build-Time: Leemos el JSON como Fuente Única de Verdad
try {
  // Verificamos y leemos el archivo
  const rawData = fs.readFileSync(dataPath, 'utf-8');
  const productos = JSON.parse(rawData);

  // Inicializamos el contenido del archivo de Cloudflare
  let redirectsContent = '# Redirecciones Perimetrales Automáticas - KalidaPresio\n';
  redirectsContent += '# Generado en Build-Time. No editar manualmente.\n\n';

  // 3. Iteración algorítmica sobre el catálogo de hardware
  productos.forEach(producto => {
    // Validamos que los datos necesarios existan antes de escribir
    if (producto.id && producto.enlace_afiliado) {
      // Estándar Cloudflare Pages: /ruta_origen  URL_destino  código_HTTP
      // Usamos 302 porque el enlace de afiliado puede cambiar en el futuro (ej. campañas temporales)
      redirectsContent += `/recomienda/${producto.id}  ${producto.enlace_afiliado}  302\n`;
    }
  });

  // 4 y 5. Escritura segura con sobrescritura automática
  // Si la carpeta public no existe, la creamos primero
  if (!fs.existsSync(outputDir)) {
    fs.mkdirSync(outputDir, { recursive: true });
  }

  // Sobrescribimos o creamos el archivo con codificación UTF-8
  fs.writeFileSync(outputPath, redirectsContent, 'utf-8');
  
  console.log(`\n✅ [KalidaPresio] Archivo _redirects inyectado en /public exitosamente.`);
  console.log(`✅ ${productos.length} rutas de afiliación protegidas (Cloaking Nativo).\n`);

} catch (error) {
  console.error('❌ [KalidaPresio] Error crítico al generar las redirecciones:', error.message);
  // Hacemos fallar el build si el archivo de datos no existe o está corrupto
  process.exit(1); 
}
